项目概述
项目名称: 宝石检测在线培训工具 (Gemstone Testing Online Training Platform)

项目类型: 教育类Web应用

核心功能:

沉浸式实验室工作台主页
三种仪器知识库(折射仪、偏光镜、分光镜)

  const currentStepConfig = refractometerInteractionFlow.steps[currentStep];
  
  if (currentStepConfig.action === elementType) {
    // 更新状态
    updateInstrumentState(elementType);
    
    // 验证步骤完成
    if (currentStepConfig.validation()) {
      setCurrentStep(currentStep + 1);
      showSuccessFeedback();
    }
  } else {
    showHint(currentStepConfig.hint);
  }

// 折射率读数验证
const validateRefractiveIndex = (
  userInput: number,
  correctValue: number,
  tolerance: number = 0.003
): boolean => {
  return Math.abs(userInput - correctValue) <= tolerance;
};

// 命名评估评分
const calculateScore = (
  difficulty: string,
  attempts: number,
  timeSpent: number // seconds
): number => {
  const basePoints = {
    'beginner': 100,
    'intermediate': 200,
    'advanced': 300
  }[difficulty];
  
  // 扣分规则
  const attemptPenalty = (attempts - 1) * 20;
  const timePenalty = timeSpent > 300 ? Math.floor((timeSpent - 300) / 60) * 5 : 0;
  
  return Math.max(basePoints - attemptPenalty - timePenalty, 10);
};

// 检测数据完整性检查
const validateDetectionData = (data: DetectionData): ValidationResult => {
  const errors: string[] = [];
  
  if (!data.refractometer?.value) {
    errors.push('缺少折射率数据');
  }
  
  if (!data.polariscope?.character) {
    errors.push('缺少光学性质数据');
  }
  
  if (!data.spectroscope?.features) {
    errors.push('缺少光谱特征数据');
  }
  
  return {
    valid: errors.length === 0,
    errors
  };
};
